SELECT * 
FROM products
LIMIT 10;

-- Этот запрос возвращает первые 10 записей из таблицы products
-- SELECT * - выбирает все столбцы
-- FROM products - указывает таблицу-источник
-- LIMIT 10 - ограничивает результат 10 записями
SELECT
*
FROM
albums;
Данный пример задает выборку всех полей (символ * обозначает «все поля»)
из таблицы albums
