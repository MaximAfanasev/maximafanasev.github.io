function factorial(n) { 
   // Именованная функция
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}
console.log(factorial(5)); // 120

рекурсия и стек 
function countdown(n) { // Рекурсивная функция
    if (n <= 0) {
        console.log("Done!");
        return;
    }
    console.log(n);
    countdown(n - 1); // Рекурсивный вызов
}
countdown(3);
// Вывод:
// 3
// 2
// 1
// Done!
