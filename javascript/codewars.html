const sorter = arr => {
	return arr.sort((a, b) => {
		const aa = a.toLowerCase()
    const bb = b.toUpperCase()
    if (aa < bb) return 1
    if (bb > aa) return -1
    return 0
	});
}

console.log(sorter(arr15))
