def remove_duplicates(lst):
    unique_lst = []
    for item in lst:
        if item not in unique_lst:
            unique_lst.append(item)
    return unique_lst

    count_dict = {}
    for item in lst:
        if item in count_dict:
            count_dict[item] += 1
        else:
            count_dict[item] = 1
    return count_dict

def bubble_sort(lst):
    n = len(lst)
    for i in range(n):
        for j in range(0, n-i-1):
            if lst[j] > lst[j+1]:
                lst[j], lst[j+1] = lst[j+1], lst[j]
return lst

# Пример использования
sample_list = [64, 34, 25, 12, 22, 11, 90]
result = bubble_sort(sample_list)
print(result)  
# Вывод: [11, 12, 22, 25, 34, 64, 90]
	
Быстрая сортировка

def quick_sort(lst):
    if len(lst) <= 1:
        return lst
    pivot = lst[len(lst) // 2]
    left = [x for x in lst if x < pivot]
    middle = [x for x in lst if x == pivot]
    right = [x for x in lst if x > pivot]
    return quick_sort(left) + middle + quick_sort(right)

# Пример использования
sample_list = [3, 6, 8, 10, 1, 2, 1]
result = quick_sort(sample_list)
print(result)  # Вывод: [1, 1, 2, 3, 6, 8, 10]
	
Сортировка вставками

def insertion_sort(lst):
    for i in range(1, len(lst)):`
        key = lst[i]
        j = i - 1
        while j >= 0 and key < lst[j]:
            lst[j + 1] = lst[j]
            j -= 1
    lst[j + 1] = key
    return lst

# Пример использования
sample_list = [12, 11, 13, 5, 6]
result = insertion_sort(sample_list)
print(result)  
# Вывод: [5, 6, 11, 12, 13]

def is_palindrome(s):
    s = s.lower().replace(" ", "")  
    # Игнорируем регистр и пробелы
    return s == s[::-1]

# Пример использования

import re

def replace_word(text, old_word, new_word):
    pattern = re.escape(old_word)  
    # Экранируем старое слово для 
    использования в регулярном выражении
    return re.sub(pattern, new_word, text)

# Пример использования
input_text = "Hello world. The world is beautiful."
result = replace_word(input_text, "world", "universe")
print(result)  
# Вывод: "Hello universe. The universe is beautiful."

def read_from_file(filename):
    try:
        with open(filename, 'r') as file:
            content = file.read()
        return content
    except FileNotFoundError:
        print(f"Файл {filename} не найден.")
        return None
    except IOError as e:
        print(f"Ошибка при 
        чтении файла {filename}: {e}")
        return None

# Пример использования

Основы ООП
1. Классы

Класс — это шаблон для создания объектов. 
Он описывает, какие свойства и методы 
будут у создаваемых объектов. 
В Python класс определяется 
с помощью ключевого слова class.
2. Наследование

Наследование позволяет одному 
классу наследовать атрибуты и методы 
другого класса. 
Это помогает повторно использовать 
код и организовывать его.
3. Инкапсуляция

Инкапсуляция — это сокрытие внутреннего 
состояния объекта и методов от внешнего 
доступа. Обычно это достигается 
с помощью модификаторов доступа.
4. Полиморфизм

Полиморфизм позволяет методам с одинаковым
 именем работать с разными типами объектов. 
 Это может быть достигнуто с помощью 
 перегрузки методов или интерфейсов.

Инкапсуляция

class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner
        self.__balance = balance  # приватный атрибут

    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount
            print(f"Депозит: {amount}. 
            Текущий баланс: {self.__balance}")
        else:
            print("Сумма вклада 
            должна быть положительной.")

    def withdraw(self, amount):
        if 0 < amount <= self.__balance:
            self.__balance -= amount
            print(f"Снятие: {amount}. 
            Текущий баланс: {self.__balance}")
        else:
            print("Недостаточно средств 
            или неверная сумма.")

    def get_balance(self):  # метод доступа
        return self.__balance

# Пример использования
account = BankAccount("Иван", 100)
account.deposit(50)  # Депозит: 50. 
Текущий баланс: 150
account.withdraw(20)  # Снятие: 20. Текущий баланс: 130
print(account.get_balance())  # Вывод: 130

    return total

def generate_report(sales_data):
    report = "Ежемесячный отчет о продажах\n"
    report += "="*30 + "\n"
    total_sales = sum(sales_data.values())

    for product, amount in sales_data.items():
        report += f"{product}: {amount}\n"

    report += f"Общие продажи: {total_sales}\n"
    return report

def save_report(report, filename):
    with open(filename, 'w', encoding='utf-8') as file:
        file.write(report)

if __name__ == '__main__':
    sales_data = {
        'Продукт A': 120,
        'Продукт B': 150,
        'Продукт C': 90,
    }
    
    report = generate_report(sales_data)
    print(report)
    save_report(report, 'sales_report.txt')
    print("Отчет сохранён в sales_report.txt")
