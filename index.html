Сортировка выбором O(n2)

  • function Sort(a) {
  •   let n = a.length
  •   for (let i=0; i<n-1; i++){
  •     let min = i
  •     for (let j=i+1; j<n; j++){
  •       if (a[j] < a[min]) min = j
  •       [a[i], a[min]] = [a[min], a[i]]
  •     }
  •   }
  •   return a
  • }

  • function bubbleSort(a) {
  •   let n = a.length;
  •   for (let i = 0; i < n - 1; i++) {
  •     for (j = 0; j < n - i; j++) {
  •       if (a[j] > a[j + 1]) {
  •         [a[j], a[j + 1]] = [a[j + 1], a[j]]
  •       }
  •     }
  •   }
  • return a
  • }
