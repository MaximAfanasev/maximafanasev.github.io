<!DOCTYPE> – объявление типа документа
<html> – корневой элемент HTML-документа
<head> – метаинформация
<title> – заголовок страницы
<body> – основное содержимое
<meta> – метаданные
<link> – подключение внешних ресурсов
<style> – встроенные CSS-стили
<script> – JavaScript
<base> – базовый URL


<!DOCTYPE html> <!-- Объявление типа документа (HTML5) -->
<html lang="ru"> <!-- Корневой элемент с указанием языка -->
<head> <!-- Метаинформация -->
    <meta charset="UTF-8"> <!-- Кодировка (обязательно для кириллицы) -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0"> <!-- Адаптивность -->
    <meta name="description" content="Пример HTML-документа"> <!-- Описание для SEO -->
    <title>Моя первая страница</title> <!-- Заголовок во вкладке браузера -->
    <base href="https://example.com/"> <!-- Базовый URL для относительных ссылок -->
    <link rel="stylesheet" href="styles.css"> <!-- Подключение внешнего CSS -->
    <link rel="icon" href="favicon.ico" type="image/x-icon"> <!-- Иконка сайта -->
    <style>
        body { 
            font-family: Arial, sans-serif; 
            margin: 0;
        }
    </style> <!-- Встроенные CSS-стили -->
</head>
<body> <!-- Основное содержимое страницы -->
    <h1>Привет, мир!</h1>
    <p>Это пример HTML-документа.</p>

    <script>
        console.log("Hello from JavaScript!"); 
    </script> <!-- Встроенный JavaScript -->
    <script src="app.js"></script> <!-- Подключение внешнего JS-файла -->
</body>
</html>

* {}                     /* Все элементы */
div {}                   /* Все <div> */
.class {}                /* Элементы с class="class" */
#id {}                   /* Элемент с id="id" */

div p {}                 /* Все <p> внутри <div> */
div > p {}               /* <p> - прямые потомки <div> */
div + p {}               /* <p> сразу после <div> */
div ~ p {}               /* Все <p> после <div> */

[title] {}               /* Элементы с атрибутом title */
[href="..."] {}          /* Точное значение атрибута */
[class^="icon-"] {}      /* Начинается с... */
[class$="-btn"] {}       /* Заканчивается на... */
[class*="error"] {}      /* Содержит подстроку */

:hover {}                /* При наведении */
:focus {}                /* Элемент в фокусе */
:first-child {}          /* Первый потомок */
:nth-child(2n) {}        /* Чётные элементы */
:not(.class) {}          /* Все, кроме .class */

::before {}              /* Добавить контент ДО */
::after {}               /* Добавить контент ПОСЛЕ */
::first-letter {}        /* Стиль первой буквы */
::selection {}           /* Выделенный текст */

:first-of-type {}        /* Первый элемент типа */
:last-of-type {}         /* Последний элемент типа */
:nth-of-type(3) {}       /* Конкретный элемент */
:only-child {}           /* Единственный потомок */

:checked {}              /* Выбранный чекбокс */
:disabled {}             /* Отключённые поля */
:valid {}                /* Корректные данные */
:placeholder-shown {}    /* Показ placeholder */
